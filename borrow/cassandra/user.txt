CREATE TABLE borrow.user(
   id timeuuid,
   cpf text,
   rg text,
   email text,
   address timeuuid,
   name text,
   description text,
   firstName text,
   middleName text,
   lastName text,
   phone text,
   updated_at timestamp,
   created_at timestamp,
   PRIMARY KEY (id, cpf, rg)
);

CREATE CUSTOM INDEX idx_user_name ON borrow.user (name) 
USING 'org.apache.cassandra.index.sasi.SASIIndex' 
WITH OPTIONS = {
'mode': 'CONTAINS', 
'analyzer_class': 'org.apache.cassandra.index.sasi.analyzer.NonTokenizingAnalyzer', 
'case_sensitive': 'false'};
